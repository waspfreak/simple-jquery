    16.Colect mouse position mousemove()/click()

              
           
             < = < >; = >
             Mouse Click : X : <input type="text" id="mClickXPos">
             Y : <input type="text" id="mClickYPos">
    

             Mouse Move : X : <input type="text" id="mMoveXPos">
             Y : <input type="text" id="mMoveYPos">
    
               
         
              
          
            $(document).click(function(e) {
              $("#mClickXPos").val(e.pageX);
              $("#mClickYPos").val(e.pageY);
            });

            $(document).mousemove(function(e) {
              $("#mMoveXPos").val(e.pageX);
              $("#mMoveYPos").val(e.pageY);
            });

    19.Simple slider img

              
           
             < = < >; = >
             <img id="slider" class="slider" src="img/1.png" alt="Img 1">
           
         
              
          
            var sliderImg = ["img/1.png", "img/2.png", "img/3.png", "img/4.png"];

            var focusImg = 1;

            $("#slider").click(function() {
              var image = sliderImg[focusImg];
              focusImg++;
              if(focusImg > 3){
                focusImg = 0;
              }
              $("#slider").attr("src", image);
            });
